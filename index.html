Learn JavaScript & HTML

Strona zawierająca podstawowe elementy HTML, CSS i JavaScript z wyjaśnieniami.

1. Podłączanie JavaScript do HTML

Aby używać JavaScript w pliku HTML, możesz dodać skrypt JavaScript bezpośrednio w sekcji <head> lub <body>, albo załadować zewnętrzny plik JavaScript.

<script src="script.js" defer></script>

2. Podstawowe składniki JavaScript

Zmienne:

// Deklaracja zmiennej
var liczba = 42;

// Można też używać let (dodane w nowszych wersjach JavaScript)
let nazwa = "Hello";

// Stałe
const PI = 3.14;

Funkcje:

function witajSwiecie() {
    alert("Witaj, świecie!");
}

Instrukcje warunkowe:

let wiek = 18;
if (wiek >= 18) {
    console.log("Jesteś pełnoletni");
} else {
    console.log("Nie jesteś pełnoletni");
}

Pętle:

// Pętla for
for (let i = 0; i < 5; i++) {
    console.log(i);
}

3. Manipulacja HTML i CSS

Pobieranie elementów HTML:

// Pobranie elementu po id
let naglowek = document.getElementById("mojNaglowek");

Manipulacja treścią i stylem:

// Zmiana treści elementu
naglowek.innerHTML = "Nowy nagłówek";

Obsługa zdarzeń:

// Obsługa zdarzenia kliknięcia
przyciski[0].addEventListener("click", function() {
    alert("Przycisk został kliknięty!");
});

4. Przydatne operacje JavaScript w HTML

Manipulacja atrybutami elementów HTML:

// Ustawienie atrybutu "src" dla obrazka o id "mojeZdjecie"
let mojeZdjecie = document.getElementById("mojeZdjecie");
mojeZdjecie.setAttribute("src", "nowe-zdjecie.jpg");

Manipulacja Klasami:

// Dodanie klasy do elementu
document.getElementById("mojElement").classList.add("nowaKlasa");

// Usunięcie klasy z elementu
document.getElementById("mojElement").classList.remove("staraKlasa");

// Sprawdzenie, czy element ma daną klasę
if (document.getElementById("mojElement").classList.contains("jakasKlasa")) {
    console.log("Element ma daną klasę");
}

Pobieranie wartości z formularza:

// Pobranie wartości z pola tekstowego o id "mojePoleTekstowe"
let mojePoleTekstowe = document.getElementById("mojePoleTekstowe");
let wartosc = mojePoleTekstowe.value;
console.log("Wprowadzona wartość: " + wartosc);

Tworzenie nowych elementów HTML:

// Stworzenie nowego paragrafu
let nowyParagraf = document.createElement("p");
nowyParagraf.textContent = "To jest nowy paragraf!";
document.body.appendChild(nowyParagraf);

Obsługa formularzy:

// Przechwytywanie zdarzenia submit formularza
let formularz = document.getElementById("mojFormularz");
formularz.addEventListener("submit", function(event) {
    event.preventDefault(); // Zapobieganie domyślnej akcji formularza
    console.log("Formularz został wysłany!");
});

Zegary i interwały czasowe:

// Wywołanie funkcji co sekundę
let interwal = setInterval(function() {
    console.log("Minęła sekunda!");
}, 1000);

// Przerwanie interwału po 5 sekundach
setTimeout(function() {
    clearInterval(interwal);
    console.log("Interwał zatrzymany po 5 sekundach.");
}, 5000);

Local Storage:

// Zapis do Local Storage
localStorage.setItem("klucz", "wartość");

// Odczyt z Local Storage
let odczytanaWartosc = localStorage.getItem("klucz");
console.log("Wartość z Local Storage: " + odczytanaWartosc);

Fetch API:

// Pobieranie danych z API
fetch('https://api.example.com/data')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error('Błąd:', error));

5. Dodatkowe Funkcje JavaScript

Geolokalizacja:

// Pobranie geolokalizacji użytkownika
navigator.geolocation.getCurrentPosition(function(position) {
    console.log("Szerokość geograficzna: " + position.coords.latitude);
    console.log("Długość geograficzna: " + position.coords.longitude);
});

Animacje CSS:

// Dodanie klasy do elementu rozpoczynającej animację
let animowanyElement = document.getElementById("animowanyElement");
animowanyElement.classList.add("animacja-start");

WebSockets:

// Utworzenie połączenia WebSocket
let socket = new WebSocket("wss://example.com");
socket.onopen = function() {
    console.log("Połączenie WebSocket otwarte.");
};

Drag and Drop:

// Implementacja obsługi przeciągania i upuszczania
let przeciaganyElement = document.getElementById("przeciaganyElement");
przeciaganyElement.addEventListener("dragstart", function(event) {
    event.dataTransfer.setData("text", "To jest przeciagany tekst.");
});

6. Manipulacja Formularzami HTML przy użyciu JavaScript

Pobieranie danych z formularza:

// Pobranie wartości pola tekstowego
let poleTekstowe = document.getElementById("poleTekstowe");
let wartoscPola = poleTekstowe.value;
console.log("Wartość pola tekstowego: " + wartoscPola);

Walidacja formularza:

// Walidacja pola email
let poleEmail = document.getElementById("poleEmail");
let wartoscEmail = poleEmail.value;
if (!wartoscEmail.includes("@")) {
    alert("Niepoprawny adres email.");
}

Dynamiczne dodawanie pól do formularza:

// Dodanie nowego pola tekstowego
let nowePoleTekstowe = document.createElement("input");
nowePoleTekstowe.type = "text";
nowePoleTekstowe.placeholder = "Nowe pole tekstowe";
document.getElementById("formularz").appendChild(nowePoleTekstowe);
